import { API_BASE_URL, API_TOKEN } from '../config';
import type {
  BrandProfile,
  BrandProfilePayload,
  Conversation,
  ConversationDetail,
  HealthResponse,
  IdeasResponse,
  PerformanceLog,
  PerformanceLogResult,
  NotificationItem,
  StreamRule,
  SchedulerJobInfo,
  GrowthSettings,
  FunnelPoint,
  ToneProfitabilityPoint,
  PlatformHistogram,
  AlertImpactResponse,
  WatchlistAnalytics,
} from '../types';

const defaultHeaders: Record<string, string> = {
  'Content-Type': 'application/json',
};

if (API_TOKEN) {
  defaultHeaders.Authorization = `Bearer ${API_TOKEN}`;
}

async function request<T>(path: string, init?: RequestInit): Promise<T> {
  const response = await fetch(`${API_BASE_URL}${path}`, {
    ...init,
    headers: {
      ...defaultHeaders,
      ...(init?.headers || {}),
    },
  });

  if (!response.ok) {
    const body = await response.text();
    throw new Error(`Request failed: ${response.status} ${response.statusText} â€” ${body}`);
  }

  return response.json() as Promise<T>;
}

export const api = {
  getHealth: () => request<HealthResponse>('/health'),

  getTopConversations: (limit = 20) =>
    request<Conversation[]>(`/conversations/top?limit=${encodeURIComponent(limit)}`),

  getConversationDetail: (platform: string, postId: string) =>
    request<ConversationDetail>(`/conversations/${encodeURIComponent(platform)}/${encodeURIComponent(postId)}`),

  getIdeas: () => request<IdeasResponse>('/ideas/today'),

  getBrandProfile: () => request<BrandProfile>('/brand/profile'),

  updateBrandProfile: (payload: BrandProfilePayload) =>
    request<{ ok: boolean }>('/brand/profile', {
      method: 'POST',
      body: JSON.stringify(payload),
    }),

  getGrowthSettings: () => request<GrowthSettings>('/growth/settings'),

  updateGrowthSettings: (payload: GrowthSettings) =>
    request<GrowthSettings>('/growth/settings', {
      method: 'POST',
      body: JSON.stringify(payload),
    }),

  getFunnel: (days = 14) => request<FunnelPoint[]>(`/analytics/funnel?days=${encodeURIComponent(days)}`),

  getToneProfitability: (min_count = 1) =>
    request<ToneProfitabilityPoint[]>(
      `/analytics/tone-profitability?min_count=${encodeURIComponent(min_count)}`,
    ),

  getPlatformHistogram: (params?: { days?: number; bins?: number }) => {
    const search = new URLSearchParams();
    if (params?.days) search.set('days', String(params.days));
    if (params?.bins) search.set('bins', String(params.bins));
    const query = search.toString();
    return request<PlatformHistogram[]>(`/analytics/platform-histogram${query ? `?${query}` : ''}`);
  },

  getPerformanceLogs: (params?: { limit?: number; tone?: string; content_type?: string }) => {
    const search = new URLSearchParams();
    if (params?.limit) search.set('limit', String(params.limit));
    if (params?.tone) search.set('tone', params.tone);
    if (params?.content_type) search.set('content_type', params.content_type);
    const query = search.toString();
    return request<PerformanceLog[]>(`/performance/logs${query ? `?${query}` : ''}`);
  },

  logPerformance: (payload: { content_type: 'reply' | 'idea'; tone?: string | null; tweet_ids: string[] }) =>
    request<PerformanceLogResult>('/performance/log', {
      method: 'POST',
      body: JSON.stringify(payload),
    }),

  getAlerts: (limit = 25) => request<NotificationItem[]>(`/alerts/recent?limit=${encodeURIComponent(limit)}`),

  getStreamRules: () => request<StreamRule[]>('/stream/rules'),

  addStreamRule: (value: string) =>
    request<{ ok: boolean }>('/stream/rules', {
      method: 'POST',
      body: JSON.stringify({ value }),
    }),

  deleteStreamRule: (id: number) =>
    request<{ ok: boolean }>(`/stream/rules/${id}`, { method: 'DELETE' }),

  getSchedulerJobs: () => request<SchedulerJobInfo[]>('/scheduler/jobs'),

  runSchedulerJob: (job_id: string) =>
    request<{ ok: boolean }>('/scheduler/run', {
      method: 'POST',
      body: JSON.stringify({ job_id }),
    }),

  toggleSchedulerJob: (job_id: string, action: 'pause' | 'resume') =>
    request<{ ok: boolean }>('/scheduler/toggle', {
      method: 'POST',
      body: JSON.stringify({ job_id, action }),
    }),

  getAlertImpact: (params?: { limit?: number; window_minutes?: number; bucket_minutes?: number }) => {
    const search = new URLSearchParams();
    if (params?.limit) search.set('limit', String(params.limit));
    if (params?.window_minutes) search.set('window_minutes', String(params.window_minutes));
    if (params?.bucket_minutes) search.set('bucket_minutes', String(params.bucket_minutes));
    const query = search.toString();
    return request<AlertImpactResponse>(`/analytics/alert-impact${query ? `?${query}` : ''}`);
  },

  getWatchlistAnalytics: (days = 14) =>
    request<WatchlistAnalytics>(`/analytics/watchlist?days=${encodeURIComponent(days)}`),
};

export type ApiClient = typeof api;
